\usepackage{minted}

\newcommand{\darkcodebox}[2] {
    \scantokens{
        \begin{tcolorbox}[
            colback=black,
            colupper=white,
            collower=white,
        ]
            {\sffamily \large #2}
            \tcblower
            \inputminted[bgcolor=black, linenos=true, style=monokai]{java}{\relcodefile{#1}}
        \end{tcolorbox}
    }
}

\newcommand{\darkcodelines}[4] {
    \scantokens{
        \begin{tcolorbox}[

        ]

}
