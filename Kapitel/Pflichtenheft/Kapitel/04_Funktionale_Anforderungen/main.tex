\begin{frame}\frametitle{Ortungsfunktion}
    \begin{columns}
        \column{0.5\textwidth}
        \begin{itemize}
            \item Standort anzeigen
            \item Keine Routenplanung
                  \begin{enumerate}
                      \item Kartenmaterial schlecht
                      \item mehrere Eingänge
                      \item kurze Routen
                      \item nicht-optimale Route schadet Benutzererfahrung stark
                  \end{enumerate}
        \end{itemize}
